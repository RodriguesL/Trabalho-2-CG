194 def rotatePolygon(point):
195  global startedPoint
196  rotatePoint = selectedPolygon.nails[0]
197 
198  startPoint = startedPoint - rotatePoint
199  anglePoint = point - rotatePoint
200 
201  inner_product = startPoint.x*anglePoint.x + startPoint.y*anglePoint.y
202 
203  len1 = math.hypot(startPoint.x, startPoint.y)
204  len2 = math.hypot(anglePoint.x, anglePoint.y)
205 
206  angle = math.acos(inner_product/(len1*len2))
207 
208  angle = angle*180/math.pi
209  angle = math.copysign(angle, anglePoint.crossProd(startPoint).z)
210  angle *= -1

266 def intersects(l1, l2):
267 
268  vector1 = Point(0,0)
269  vector2 = Point(0,0)
270 
271  vector1.x = l1.p2.x - l1.p1.x
272  vector1.y = l1.p2.y - l1.p1.y
273 
274  vector2.x = l2.p2.x - l2.p1.x
275  vector2.y = l2.p2.y - l2.p1.y
276 
277  s = (-vector1.y * (l1.p1.x - l2.p1.x) + vector1.x * (l1.p1.y - l2.p1.y)) / (-vector2.x * vector1.y + vector1.x * vector2.y)
278  t = ( vector2.x * (l1.p1.y - l2.p1.y) - vector2.y * (l1.p1.x - l2.p1.x)) / (-vector2.x * vector1.y + vector1.x * vector2.y)
279 
280  if (s >= 0 and s <= 1 and t >= 0 and t <= 1):
281  return True
282  return False
